# dfr get-month

Gets month from date

# Signature

> dfr get-month

# Examples

Returns month from a date

> let dt = ('2020-08-04T16:39:18+00:00' | into datetime -z 'UTC');
    let df = ([$dt $dt] | dfr to-df);
    $df | dfr get-month
